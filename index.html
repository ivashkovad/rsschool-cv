Code Example

Codewars Task: Digital Root (calculates the recursive sum of all digits in a number until a single-digit result is reached)

function digitalRoot(n) {
  if (n < 10) return n;

  const sum = n
    .toString()
    .split('')
    .reduce((acc, digit) => acc + Number(digit), 0);

  return digitalRoot(sum);
}
